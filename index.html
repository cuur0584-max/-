// --- V28 智慧校正邏輯片段 --- function handleRecord(type) { const now = Date.now(); currentRecType = type; recognitionMode = 'seconds'; // 1. 尋找或建立路口紀錄 if (selectedIndex === -1) { records.push({ sn: globalCounter++, name: "", lat: userPos.lat, lon: userPos.lon, heading: currentHeading, redDur: 0, greenDur: 0, redSync: 0, greenSync: 0 }); selectedIndex = records.length - 1; } const rec = records[selectedIndex]; // 2. 自動校正邏輯 (核心！) if (type === 'green' && rec.redSync > 0) { // 如果現在按綠燈，且之前有按紅燈 const measuredRedDur = now - rec.redSync; // 如果測得的紅燈時間與紀錄的 redDur 誤差超過 3 秒 if (rec.redDur > 0 && Math.abs(measuredRedDur - rec.redDur) > 3000) { const correctedSec = Math.round(measuredRedDur / 1000); rec.redDur = correctedSec * 1000; // 自動修正 speak(`已自動修正紅燈為 ${correctedSec} 秒`); } else if (rec.redDur === 0) { // 如果原本沒錄過秒數，這一次按鍵自動當作第一次學習 rec.redDur = Math.round(measuredRedDur / 1000) * 1000; speak(`學習完成，紅燈 ${Math.round(measuredRedDur / 1000)} 秒`); } } // 3. 更新同步點 if (type === 'red') rec.redSync = now; else rec.greenSync = now; saveData(); updateUI(); document.getElementById(type + '-btn').classList.add('listening'); document.getElementById('sound-main').play(); // 4. 開啟語音辨識 (讓你還是可以手動輸入) setTimeout(() => startRecognition('seconds'), 400); }
